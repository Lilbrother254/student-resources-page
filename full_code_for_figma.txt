        const resourcesGrid = document.querySelector('.resources-grid');
        const modal = document.getElementById('resourceModal');
        const modalBody = document.getElementById('modalBody');
        const closeBtn = document.querySelector('.close-btn');
        const navItems = document.querySelectorAll('.nav-item');

        function displayResources() {
            resourcesGrid.innerHTML = '';
            resources.forEach(resource => {
                const card = document.createElement('a');
                card.href = "#";
                card.className = `resource-card ${resource.color}`;
                card.innerHTML = `
                    <i class="icon ${resource.icon}"></i>
                    <span class="text">${resource.title}</span>
                `;
                card.addEventListener('click', (e) => {
                    e.preventDefault();
                    modalBody.innerHTML = resource.modalContent;
                    modal.style.display = 'flex'; // Use flex to center modal
                });
                resourcesGrid.appendChild(card);
            });
        }

        // Close modal when clicking on close button or outside modal
        closeBtn.addEventListener('click', () => {
            modal.style.display = 'none';
        });

        window.addEventListener('click', (e) => {
            if (e.target == modal) {
                modal.style.display = 'none';
            }
        });

        // Handle active navigation item
        navItems.forEach(item => {
            item.addEventListener('click', () => {
                navItems.forEach(nav => nav.classList.remove('active'));
                item.classList.add('active');
            });
        });

        // Initial display of resources
        displayResources();
    </script>
